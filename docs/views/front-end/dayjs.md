# dayjs常用功能总结

```
<body>     <script src="https://unpkg.com/dayjs"></script>    <script>    	console.log(0, dayjs(1578019476000).format('YYYY-MM-DD HH:mm:ss'));  // "2020-01-03 10:44:36"        console.log(1,dayjs().format());  // "2019-09-16T21:03:47+08:00"        console.log(2,dayjs().format('YYYY-MM-DD')); // "2019-09-16"        // 时间往前推        console.log(3,dayjs().subtract(7, 'day').format('YYYY-MM-DD')); // "2019-09-09"        console.log(4,dayjs().subtract(7, 'year').format('YYYY-MM-DD')); // "2012-09-16"        console.log(5,dayjs().subtract(7, 'month').format('YYYY-MM-DD')); // "2019-09-23"        console.log(15, dayjs().subtract(1, 'month').format('YYYY-MM-DD')); // '2019-08-09'        console.log(13, dayjs().subtract(6, 'month').subtract(18, 'day').format('YYYY-MM-DD'));        console.log(14, dayjs().subtract(6, 'month').subtract(17, 'day').subtract(1, 'month').format('YYYY-MM-DD'))        // 时间往后推        console.log(6,dayjs().add(7, 'day').format('YYYY-MM-DD')); // "2019-09-23"        console.log(7,dayjs().add(7, 'year').format('YYYY-MM-DD')); // "2026-09-16"        // 时间差        console.log(8, dayjs('2018-06-08').diff(dayjs('2017-06-01'),'years')); // 1        console.log(9, dayjs('2018-06-08').diff(dayjs('2017-06-01'),'day')); // 372         console.log(10, dayjs('2018-06-08').diff(dayjs('2017-06-01'),'hour')); // 8928        // 获取时间戳        console.log(11, dayjs('2018-06-08').valueOf()); //毫秒    </script> </body>
```
